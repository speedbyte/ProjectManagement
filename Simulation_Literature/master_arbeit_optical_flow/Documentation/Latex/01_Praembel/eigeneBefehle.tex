% Anführungszeichen für wörtliche Zitate
\newcommand{\Zitat}[1]{
\glqq#1\grqq\xspace
}


% Blockzitat
\newcommand{\BZitat}[1]{
\begin{quote}
    \small
    #1
\end{quote}
}

% englisches Blockzitat
\newcommand{\EngBZitat}[1]{
\begin{quote}
    \small
    \begin{otherlanguage}{english}
        #1
    \end{otherlanguage}
\end{quote}
}

\newcommand{\uusB}[1]{%
'#1'\xspace
}
