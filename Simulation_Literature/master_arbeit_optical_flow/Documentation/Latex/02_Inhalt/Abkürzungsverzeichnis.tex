\section*{List of abbreviations}

\begin{itemize*}
	\item[anv\_propX] angular velocity of propeller X
	\item[aphi] angular acceleraction in direction of phi
	\item[apsi] angular acceleraction in direction of psi
	\item[AT] sum of all rotary speeds anv\_propX
	\item[atheta] angular acceleraction in direction of theta
	\item[ax] acceleration in direction of x of bodyframe
	\item[ay] acceleration in direction of y of bodyframe
	\item[az] acceleration in direction of z of bodyframe
	\item[CL] Closed Loop
	\item[F\_propX] resulting force of propeller X
	\item[HIL] Hardware In the Loop
	\item[IMU] Inertial measurement unit
	\item[KR] vector holding feedback factors
	\item[Kv] vector holding pre-intensification factors
	\item[MIMO] Multiple Input Multiple Output
	\item[MU] measurement unit
	\item[phi] roll angle around the x-axis of bodyframe
	\item[phi\_set] set point of phi
	\item[pos\_vel\_acc] positions\_velocities\_accelerations
	\item[psi] yaw angle around the z-axis of bodyframe
	\item[rphi] angular rate of phi (roll rate)
	\item[rpsi] angular rate of psi (yaw rate)
	\item[rpsi\_set] set point of rpsi
	\item[rtheta] angular rate of theta (pitch rate)
	\item[SIMO] Single Input Multiple Output
	\item[SISO] Single Input Single Output
	\item[theta] pitch angle around the y-axis of bodyframe
	\item[theta\_set] set point of theta
	\item[thrust\_set] set point of thrust
	\item[U1] sum of all propeller forces F\_propX (variable in Simulink)
	\item[U2] angular acceleration in direction of phi (variable in Simulink)
	\item[U3] angular acceleration in direction of theta (variable in Simulink)
	\item[U4] angular acceleration in direction of psi (variable in Simulink)
\end{itemize*}
